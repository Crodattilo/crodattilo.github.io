@echo off
color F1
echo Gestore PC
echo.

:start
cls
echo Gestore PC
echo.
echo Digita un comando: 
set/p A=
if %A% == help goto :help
if %A% == version goto :version
if %A% == power_off goto :power_off
if %A% == reboot goto :reboot
if %A% == cmd goto :cmd
if %A% == control_panel goto :control_panel
if %A% == task_manager goto :task_manager
if %A% == manage_files goto :manage_files
if %A% == pc_info goto :pc_info
if %A% == new_text goto :new_text
if %A% == browser goto :browser
if %A% == color goto :color
if %A% == close goto :close

:help
cls
echo.
echo Hai bisogno di aiuto? Ecco la lista di tutti i comandi:
echo.
echo 1)  VERSION.....................per vedere la versione di questo programma
echo 2)  POWER_OFF...................per spegnere il PC
echo 3)  REBOOT......................per riavviare il PC
echo 4)  PC_INFO.....................per vedere le info del PC in uso
echo 5)  BROWSER.....................per eseguire il browser predefinito
echo 6)  MANAGE_FILES................per eseguire il file manager
echo 7)  TASK_MANAGER................per eseguire Gestione Attività
echo 8)  CONTROL_PANEL...............per eseguire il pannello di controllo
echo 9)  CMD.........................per eseguore il Prompt Dei Comandi di Windows
echo 10) NEW_TEXT....................per creare un documento di testo in formato txt
echo 11) COLOR.......................per sciegliere i colori dell'interfaccia
echo 12) CLOSE.......................per arrestare Gestore PC
echo.
pause
goto :start

:version
cls
echo.
echo PC_Manager, italian version 1.0 (beta), windows OS required.
echo.
pause
goto :start

:power_off
cls
echo.
echo Sei sicuro di voler spegnere il tuo PC?
pause
shutdown -s -t 10
echo Il computer verrà spento in 10 secondi...
pause
exit

:reboot
cls
echo.
echo Sei sicuro di voler riavviare il tuo PC?
pause
shutdown -r -t 10
echo Il computer verrà riavviato in 10 secondi...
pause
exit

:cmd
cls
echo.
start cmd.exe
goto :start

:control_panel
cls
echo.
control
goto :start

:task_manager
cls
echo.
start taskmgr.exe
goto :start

:manage_files
cls
echo.
start explorer.exe
goto :start

:pc_info
cls
echo.
start msinfo32.exe
goto :start

:new_text
cls
echo.
start notepad.exe
goto :start

:close
cls
echo.
pause
exit

:browser
cls
echo.
start www.google.com
goto :start

:color
cls
echo.
echo Sciegli un tema,
echo CHIARO o SCURO:
set/p B=
if %B% == chiaro goto :light_theme
if %B% == scuro goto :dark_theme

:light_theme
cls
color F1
goto :start

:dark_theme
cls
color 1F
goto :start
